            h1,h2,h3,h4,h5,h6 {
                line-height:1;font-family:Arial,sans-serif;margin:1.4em 0 0.8em;
            }
            h1{font-size:1.8em;}
            h2{font-size:1.6em;}
            h3{font-size:1.4em;}
            h4{font-size:1.2em;}
            h5,h6{font-size:1em;}

            .typo p, .typo pre, .typo ul, .typo ol, .typo dl, .typo form, .typo hr {
                margin:1em 0 0.6em;
            }
